






SELECT qr_requests.schedule_date, 
sum(qr_requests.no_of_packages) as total_packages,
sum(qr_requests.no_of_master) as total_masters, entity_field_data.data 
FROM qr_requests    
JOIN qr_entity_data ON  qr_entity_data.request_id = qr_requests.request_id and qr_requests.schedule_date BETWEEN DATE_SUB( CURDATE() ,INTERVAL 30 DAY ) AND DATE_SUB(CURDATE(), INTERVAL 1 DAY)  
join entity_data on entity_data.entity_data_id = qr_entity_data .entity_data_id
join entity_field_data on entity_field_data.entity_data_id =entity_data.entity_data_id
join entity_fields on entity_fields.field_id = entity_field_data.entity_field_id and entity_fields.name = 'Warehouse Name'
join entities on entities.entity_id = entity_data.entity_id and entities.name = 'Warehouse'
group by entity_field_data.data, qr_requests.schedule_date order by qr_requests.schedule_date
